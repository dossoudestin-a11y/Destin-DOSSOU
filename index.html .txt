<script>
const promptEl = document.getElementById('prompt');
const outputEl = document.getElementById('output');
const btnEl = document.getElementById('generateBtn');

function setText(text){
  promptEl.value = text;
}

function genererSite(){
  const idee = promptEl.value.toLowerCase();
  if(!idee){alert('Décris ton site d’abord Destin 😅');return}
  
  btnEl.disabled = true;
  btnEl.innerText = 'Génération en cours...';
  outputEl.innerText = 'Destin IA réfléchit...';

  setTimeout(() => {
    let html = '';

    if(idee.includes('portfolio') || idee.includes('cv')){
      html = genererPortfolio(idee);
    } else if(idee.includes('boutique') || idee.includes('vendre') || idee.includes('chaussure')){
      html = genererBoutique(idee);
    } else if(idee.includes('jeu') || idee.includes('crash') || idee.includes('avion')){
      html = genererJeu(idee);
    } else if(idee.includes('coiffure') || idee.includes('salon') || idee.includes('restaurant')){
      html = genererVitrine(idee);
    } else {
      html = genererBase(idee);
    }

    outputEl.innerText = html;
    btnEl.disabled = false;
    btnEl.innerText = '✨ Régénérer';
  }, 1500);
}

function genererPortfolio(idee){
  const nom = idee.includes('destin') ? 'Destin' : 'Ton Nom';
  const couleur = idee.includes('vert') ? '#00ff88' : '#4f46e5';
  return `<!DOCTYPE html>
<html lang="fr">
<head>
<meta charset="UTF-8">
<title>${nom} - Portfolio</title>
<style>
body{font-family:Arial;background:#0a0a0a;color:#eee;text-align:center;padding:50px}
h1{color:${couleur};font-size:48px}
.btn{background:${couleur};color:#000;padding:15px 30px;border-radius:10px;text-decoration:none;font-weight:bold}
</style>
</head>
<body>
  <h1>Salut, c'est ${nom}</h1>
  <p>Développeur Web | Basé à Abomey-Calavi</p>
  <a href="https://wa.me/229" class="btn">Me contacter sur WhatsApp</a>
</body>
</html>`;
}

function genererBoutique(idee){
  const couleur = idee.includes('rouge') ? '#ef4444' : '#f59e0b';
  return `<!DOCTYPE html>
<html lang="fr">
<head>
<meta charset="UTF-8">
<title>Boutique</title>
<style>
body{font-family:Arial;background:#fff;color:#111;text-align:center;padding:40px}
h1{color:${couleur}}
.produit{border:1px solid #ddd;padding:20px;border-radius:15px;margin:20px auto;max-width:300px}
.btn{background:${couleur};color:#fff;padding:12px 24px;border-radius:8px;text-decoration:none;display:inline-block}
</style>
</head>
<body>
  <h1>Ma Boutique</h1>
  <div class="produit">
    <h3>Produit 1</h3>
    <p>5.000 FCFA</p>
    <a href="https://wa.me/229?text=Je veux commander" class="btn">Commander sur WhatsApp</a>
  </div>
</body>
</html>`;
}

function genererJeu(idee){
  return `<!DOCTYPE html>
<html lang="fr">
<head>
<meta charset="UTF-8">
<title>Crash Plane</title>
<style>
body{background:#000;color:#fff;text-align:center;font-family:Arial}
#avion{font-size:60px;animation:fly 2s infinite}
@keyframes fly{50%{transform:translateY(-30px)}}
.btn{background:#00ff88;color:#000;padding:15px 30px;border:none;border-radius:10px;font-size:18px;margin-top:30px}
</style>
</head>
<body>
  <h1>🚀 Crash Plane</h1>
  <div id="avion">✈️</div>
  <p>Multiplicateur: x1.00</p>
  <button class="btn" onclick="alert('Jeu lancé !')">Jouer</button>
</body>
</html>`;
}

function genererVitrine(idee){
  const couleur = idee.includes('rose') ? '#ec4899' : '#8b5cf6';
  return `<!DOCTYPE html>
<html lang="fr">
<head>
<meta charset="UTF-8">
<title>Mon Salon</title>
<style>
body{font-family:Arial;background:#fafafa;color:#111}
header{background:${couleur};color:#fff;text-align:center;padding:40px}

</style>
</head>
<body>
  <header><h1>Salon de Coiffure</h1><p>Abomey-Calavi</p></header>
  <div class="container" style="max-width:600px;margin:auto;padding:20px">
    <div class="card"><h3>Coiffure Homme</h3><p>2.000 FCFA</p></div>
    <div class="card"><h3>Coiffure Dame</h3><p>3.500 FCFA</p></div>
  </div>
</body>
</html>`;
}

function genererBase(idee){
  return `<!DOCTYPE html>
<html lang="fr">
<head>

</head>
<body>
  <h1>Mon Site</h1>
  <p>${idee}</p>
</body>
</html>`;
}

function copierCode(){
  navigator.clipboard.writeText(outputEl.innerText);
  alert('Code copié ! Colle-le dans un fichier .html et ouvre-le 👌');
}
